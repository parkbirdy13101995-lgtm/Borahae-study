<script>

let questions = [
  {
    q: "When did BTS debut?",
    options: ["2013", "2015", "2012", "2016"],
    correct: "2013"
  },
  {
    q: "Which is NOT a BTS song?",
    options: ["Dynamite", "Butter", "Shape of You", "Spring Day"],
    correct: "Shape of You"
  }
];

let current;
let time;
let studyTime;
let trapMode;

function startQuiz(sub) {
  document.getElementById("start").classList.add("hidden");
  document.getElementById("quiz").classList.remove("hidden");
  document.getElementById("subject").innerText = "Study: " + sub;

  loadQuestion();
}

function loadQuestion() {
  current = questions[Math.floor(Math.random()*questions.length)];
  trapMode = Math.random() < 0.5;

  document.getElementById("question").innerText = current.q;
  document.getElementById("mode").innerText = trapMode ? "⚠ Choose WRONG answer" : "Choose correct answer";

  let optionsHTML = "";
  current.options.forEach(opt => {
    optionsHTML += `<button onclick="answer('${opt}')">${opt}</button>`;
  });
  document.getElementById("options").innerHTML = optionsHTML;

  time = 10;
  let t = setInterval(()=>{
    time--;
    document.getElementById("timer").innerText = "Time: " + time;

    if(time<=0){
      clearInterval(t);
      wrong();
    }
  },1000);
}

function answer(choice){
  let isCorrect = choice === current.correct;

  if((!trapMode && isCorrect) || (trapMode && !isCorrect)){
    alert("Oh? Not bad 😏");
    loadQuestion();
  } else {
    wrong();
  }
}

function wrong(){
  document.getElementById("quiz").classList.add("hidden");
  document.getElementById("study").classList.remove("hidden");

  document.getElementById("studyText").innerText = "Focus now. Come back stronger.";

  studyTime = 1500;

  let s = setInterval(()=>{
    studyTime--;
    document.getElementById("studyTimer").innerText = "Time: " + studyTime;

    if(studyTime<=0){
      clearInterval(s);
      alert("Time’s up 💜");
      location.reload();
    }
  },1000);
}

function saveEntry(){
  let text = document.getElementById("entry").value;
  localStorage.setItem("diary", text);
  document.getElementById("saved").innerText = "Saved 💜";
}

function postGoal(){
  let goal = document.getElementById("goal").value;
  let div = document.createElement("p");
  div.innerText = goal + " 🔥";
  document.getElementById("goalsList").appendChild(div);
}
